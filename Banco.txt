    );

DELIMITER $
CREATE PROCEDURE novoCurso(
Nome_do_Curso VARCHAR(50),
Descricao VARCHAR(70),
Quantidade_Alunos INT
)
BEGIN
	INSERT INTO Curso VALUES (null, Nome_do_Curso, Descricao, Quantidade_Alunos);
    END$    
CREATE PROCEDURE selecionarCurso()
BEGIN    
	SELECT * FROM Curso;
    END$
DELIMITER ;

DELIMITER $
CREATE PROCEDURE gerarEmail(
IN Nome VARCHAR(20),
IN Data_de_Nascimento DATE,
OUT Email VARCHAR(30)
)
BEGIN
    SET Email = CONCAT(Nome, '.', DATE_FORMAT(Data_de_Nascimento, '%Y%m%d'), '@dominio.com');
    END$
DELIMITER ;
